Написать SQL запрос:

- выбор пользователя у которых 6 детей;
> SELECT * FROM `пользователи` WHERE `количество_детей` = '6'

- предоставить перечень услуг с просроченными запросами;
> SELECT * FROM `услуги` WHERE `статус_запроса` = 'просрочен'
или
Используем условие: пусть запрос является просроченным, если последнее изменение статуса было осуществлено 3 дня назад
> $дата_просроченного_запроса = date("M-d-Y", mktime(0, 0, 0, date('m'), date('d') - 3, date('Y')));
> $mysql = new mysqli($хост,$логин,$пароль,$название_базы_данных);
> $mysql->query(SELECT * FROM `услуги` WHERE `дата_запроса` = '$дата_просроченного_запроса');

- Другие запросы.
Создание объекта
> INSERT INTO `объекты` (`параметр_1`, `параметр_2`, `параметр_3`) VALUES ( '$значение_параметра_1', '$значение_параметра_2', '$значение_параметра_3')
Изменение параметров объекта
> UPDATE `объекты` SET `параметр_1` = '$новое_значение_параметра_1' WHERE `идентификатор_объекта` = '$идентификатор_объекта'
Удаление объекта
> DELETE FROM `объекты` WHERE `идентификатор_объекта` = '$идентификатор_объекта'
